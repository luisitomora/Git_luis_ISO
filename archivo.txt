while getopts "x:a:c:v:" opcion; do
    case $opcion in
        x)
            TARFILE="$OPTARG"
            echo "Extrayendo $TARFILE..."
            tar -xf "$TARFILE"
            ;;
        a)
            TARFILE="$OPTARG"
            echo "Añadiendo todos los .txt a $TARFILE..."
            find / -type f -name "*.txt" 2>/dev/null | xargs tar -rf "$TARFILE"
            ;;
        c)
            TARFILE="$OPTARG"
            echo "Creando tar $TARFILE desde HOME..."
            tar -cf "$TARFILE" "$HOME"
            ;;
        v)
            TARFILE="$OPTARG"
            echo "Contenido del tar $TARFILE:"
            tar -tf "$TARFILE"
            ;;
        *)
            echo "Opción inválida"
            exit 1
            ;;
    esac
done
